60 /// lexicographical eq comparison of str_subscriptables
61 [[nodiscard]] constexpr inline bool
62 eq(const str_subscriptable auto &a, const str_subscriptable auto &b) noexcept {
63  size_t i = 0;
64  for (; a[i] and b[i]; ++i) {
65  if (a[i] != b[i])
66  return false;
67  }
68  return a[i] == b[i];

72 [[nodiscard]] constexpr inline bool
73 eq(const str_range auto &a, const str_subscriptable auto &b) noexcept {
74  auto &&ita = begin(a);
75  size_t ib = 0;
76  for (; !(ita == end(a) || !*ita) and b[ib]; ++ita, ++ib) {
77  if (*ita != b[ib])

79  }
80  return (ita == end(a) || !*ita) and !b[ib];

89 /// lexicographical eq comparison of str_ranges
90 [[nodiscard]] constexpr inline bool
91 eq(const str_range auto &a, const str_range auto &b) noexcept {
92  auto &&ita = begin(a);
93  auto &&itb = begin(b);
94  for (; !(ita == end(a) || !*ita) and !(itb == end(b) || !*itb);
95  ++ita, ++itb) {
96  if (*ita != *itb)
97  return false;
98  }
99  return (ita == end(a) || *ita == 0) and (itb == end(b) || *itb == 0);
100 }
